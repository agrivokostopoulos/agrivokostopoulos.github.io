\section{Research Interests}

\cventry{}{Core Themes}{}{}{}{Sovereign Debt; Financial Markets; Political Institutions; First Age of Globalisation; Text-as-Data; LLMs; History of Economic Thought}

\cventry{}{Working Papers}{}{}{}{
\begin{itemize}
\item ``Made in Greece or Echoed from Europe? Sentiment and Awareness Among International Partners During the 1931--1932 Greek Sovereign Debt Crisis'' (draft). Employs a novel LLM-based text-as-data approach on the Tsouderos Archive (private correspondence of the Bank of Greece Governor) to map sentiment and topic trends in communications with foreign officials. Identifies two structural breaks (June and September 1931) via change point analysis; SVAR shows significant negative correlation between archive sentiment and Greek bond yields.
\item ``From War Finance to Fiscal Supervision: Seniority Structures and Debt Restructuring in the Ottoman Empire, Egypt, and Greece, 1870--1907'' (draft). First comparative empirical analysis of supervisory regimes (OPDA, Caisse de la Dette, IFC) using bond-level London Stock Exchange data. DiD estimates: $-1.39$ pp (Ottoman), $-0.42$ pp (Egypt), $-0.36$ pp (Greece), all $p<0.001$. Greek case shows yield spread inversion ($p=0.035$) driven by state-contingent coupons; proposes minimal extension to Ho--Ritschl (2024) seniority reversal model.
\item ``Permeable Patronage: Fiscal Crises and Domain-Specific Defection in the Greek Parliament, 1875--1905'' (draft). Tests whether fiscal shocks produce programmatic legislative transitions using 755 nominal roll call votes (132,246 MP--vote observations, 1,049 MPs) linked to a biographical registry and classified by policy domain across eight political shocks. After the 1893 default, Peloponnesian MPs defect 7.3 pp more on taxation ($p=0.006$) but not on non-fiscal votes. Adapts Krehbiel's identification problem to clientelist legislatures via $\pi$-pattern/$\pi$-binding decomposition.
\item ``Between Civilisation and Constitution: Greek Parliamentary Debates on Gold Standard Adoption, 1875--1885'' (draft, prepared for Mary Morgan Retirement Workshop, June 2026). Reconstructs parliamentary debates using newly digitised Hellenic Parliament proceedings. Argues the gold standard operated as a ``model'' (Morgan's sense) whose authority depended on narrative embedding; the central claim is \textit{misapplication} rather than wrong theory. Maps eleven parliamentarians across constitutional, distributional, regional, and technical registers; analyses absence of bimetallism as counter-narrative.
\end{itemize}}

\cventry{}{Work in Progress}{}{}{}{
\begin{itemize}
\item ``Sterling by Design: Contractual Standardisation and Currency Dominance on the London Stock Exchange, 1869--1929''. Security-level panel of 957 sovereign debt instruments (IMM). Sterling's share of income flows rose from 79\% to 99\% as multi-currency clauses, bullion clauses, and non-sterling denominations were eliminated. Bai--Perron breaks in late 1870s--1880s precede periphery gold adoption; captive borrowers paid 0.90 pp yield premium over outside-option borrowers.
\item ``Why Did Depositors Run? Withdrawals in the German Banking Crisis of 1931'' (with Philip Fliers, Queen's University Belfast). Bimonthly bank-level balance sheets (1925--1935). Severe withdrawals were widespread, highly predictable up to two years prior, and differed systematically across maturities. Withdrawals accompanied by substantial deposit reallocation across banks.
\end{itemize}}
